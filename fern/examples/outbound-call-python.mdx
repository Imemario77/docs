## Overview

Make outbound calls programmatically using various SDKs and languages. This guide shows you how to initiate calls, configure assistants, and handle responses across different platforms.

## Prerequisites

* A [Vapi account](https://dashboard.vapi.ai/) with API access
* An assistant ID or inline assistant configuration
* A phone number (optional - Vapi can provide one)

<Tabs>
  <Tab title="Dashboard">
    Create and manage outbound calls through the Vapi Dashboard:

    1. Navigate to **Calls** in your [Vapi Dashboard](https://dashboard.vapi.ai/)
    2. Click **Create Call**
    3. Configure call settings:
       - **Type**: Outbound Call
       - **Assistant**: Select your assistant or create inline
       - **Customer Number**: Enter the phone number to call
       - **Phone Number**: Select your outbound phone number
    4. Click **Start Call** to initiate the outbound call
    5. Monitor call progress and view real-time transcript
    6. Access call recordings and analysis after completion

    <Note>
      Dashboard calls are great for testing and one-off calls. Use the API for automated outbound campaigns.
    </Note>

    ```typescript
    import Vapi from '@vapi-ai/web';

    interface VapiCallerConfig {
      publicApiKey: string;
      assistantId: string;
    }

    function createVapiCaller(config: VapiCallerConfig) {
      const vapi = new Vapi(config.publicApiKey);
      let isConnected = false;

      // Setup event listeners
      vapi.on('call-start', () => {
        isConnected = true;
        console.log('Web call started');
      });

      vapi.on('call-end', () => {
        isConnected = false;
        console.log('Web call ended');
      });

      vapi.on('message', (message) => {
        if (message.type === 'transcript') {
          console.log(`${message.role}: ${message.transcript}`);
        } else if (message.type === 'function-call') {
          console.log('Function called:', message.functionCall.name);
        }
      });

      vapi.on('error', (error) => {
        console.error('Call error:', error);
      });

      return {
        startCall: () => {
          if (!isConnected) {
            vapi.start(config.assistantId);
          }
        },
        endCall: () => {
          if (isConnected) {
            vapi.stop();
          }
        },
        isConnected: () => isConnected
      };
    }

    // Usage
    const vapiCaller = createVapiCaller({
      publicApiKey: 'YOUR_PUBLIC_API_KEY',
      assistantId: 'YOUR_ASSISTANT_ID'
    });

    // Start browser-based voice interaction
    vapiCaller.startCall();
    ```

    <Note>
      Web SDK creates browser-based voice interactions. For phone calls to external numbers, use Server SDK APIs.

    Monitor outbound calls in real-time:

    ```typescript
    import Vapi from '@vapi-ai/web';

    function createVapiCallMonitor(publicApiKey: string) {
      const vapi = new Vapi(publicApiKey);

      // Note: Web SDK cannot initiate outbound phone calls
      // This monitors web-based voice interactions only
      
      function monitorCall(assistantId: string) {
        vapi.start(assistantId);
        
        vapi.on('call-start', () => {
          console.log('Web-based call monitoring started');
        });

        vapi.on('message', (message) => {
          if (message.type === 'transcript') {
            console.log(`${message.role}: ${message.transcript}`);
          } else if (message.type === 'function-call') {
            console.log('Function called:', message.functionCall.name, message.functionCall.parameters);
          }
        });

        vapi.on('call-end', () => {
          console.log('Web-based call monitoring ended');
        });

        vapi.on('error', (error) => {
          console.error('Call monitoring error:', error);
        });
      }

      return {
        monitorCall: monitorCall,
        stop: () => vapi.stop()
      };
    }

    // Usage
    const callMonitor = createVapiCallMonitor('YOUR_PUBLIC_API_KEY');
    callMonitor.monitorCall('YOUR_ASSISTANT_ID');
    ```

    ```python
    import asyncio
    import aiohttp
    from typing import List, Dict

    class VapiBulkCaller:
        def __init__(self, api_key: str):
            self.api_key = api_key
            self.base_url = "https://api.vapi.ai"
            
        async def create_call(self, session: aiohttp.ClientSession, call_data: Dict) -> Dict:
            headers = {
                "Authorization": f"Bearer {self.api_key}",
                "Content-Type": "application/json"
            }
            
            async with session.post(
                f"{self.base_url}/call",
                headers=headers,
                json=call_data
            ) as response:
                if response.status == 201:
                    return await response.json()
                else:
                    error_text = await response.text()
                    print(f"Failed to create call: {error_text}")
                    return None

        async def create_bulk_calls(self, call_list: List[Dict]) -> List[Dict]:
            async with aiohttp.ClientSession() as session:
                tasks = [
                    self.create_call(session, call_data) 
                    for call_data in call_list
                ]
                results = await asyncio.gather(*tasks, return_exceptions=True)
                return [r for r in results if r is not None]

    # Usage example
    async def main():
        caller = VapiBulkCaller("YOUR_API_KEY")
        
        # Prepare multiple calls
        calls = [
            {
                "phoneNumberId": "YOUR_PHONE_NUMBER_ID",
                "customer": {"number": "+14151111111", "name": "Customer 1"},
                "assistantId": "YOUR_ASSISTANT_ID"
            },
            {
                "phoneNumberId": "YOUR_PHONE_NUMBER_ID", 
                "customer": {"number": "+14152222222", "name": "Customer 2"},
                "assistantId": "YOUR_ASSISTANT_ID"
            },
            # Add more calls as needed
        ]
        
        results = await caller.create_bulk_calls(calls)
        print(f"Created {len(results)} calls successfully")
        
        for call in results:
            print(f"Call ID: {call['id']}, Status: {call['status']}")

    # Run the bulk call creation
    asyncio.run(main())
    ```

  <Tab title="Call Monitoring (TypeScript)">
    ```typescript
    import { Vapi } from "@vapi-ai/server-sdk";

    class VapiCallMonitor {
      private vapi: Vapi;

      constructor(apiKey: string) {
        this.vapi = new Vapi({ apiKey });
      }

      async createAndMonitorCall(callData: any): Promise<void> {
        try {
          // Create the call
          const call = await this.vapi.calls.create(callData);
          console.log(`Call created: ${call.id}`);

          // Monitor call status
          await this.monitorCallStatus(call.id);
          
          // Get final call analysis
          const finalCall = await this.vapi.calls.get(call.id);
          console.log("Call completed!");
          console.log(`Duration: ${finalCall.endedAt - finalCall.startedAt}ms`);
          
          if (finalCall.analysis) {
            console.log("Call Analysis:");
            console.log(`Summary: ${finalCall.analysis.summary}`);
            console.log(`Success: ${finalCall.analysis.successEvaluation}`);
          }
          
        } catch (error) {
          console.error("Error in call monitoring:", error);
        }
      }

      private async monitorCallStatus(callId: string): Promise<void> {
        const maxAttempts = 60; // Monitor for up to 5 minutes
        let attempts = 0;

        return new Promise((resolve, reject) => {
          const checkStatus = async () => {
            try {
              const call = await this.vapi.calls.get(callId);
              console.log(`Call ${callId} status: ${call.status}`);

              if (call.status === 'ended') {
                resolve();
                return;
              }

              attempts++;
              if (attempts >= maxAttempts) {
                reject(new Error('Call monitoring timeout'));
                return;
              }

              // Check again in 5 seconds
              setTimeout(checkStatus, 5000);
            } catch (error) {
              reject(error);
            }
          };

          checkStatus();
        });
      }
    }

    // Usage
    const monitor = new VapiCallMonitor("YOUR_API_KEY");

    monitor.createAndMonitorCall({
      phoneNumberId: "YOUR_PHONE_NUMBER_ID",
      customer: {
        number: "+14151231234",
        name: "Monitored Customer"
      },
      assistantId: "YOUR_ASSISTANT_ID"
    });
    ```
